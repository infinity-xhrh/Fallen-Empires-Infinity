# FEI / GateGrander scripted effects.
# Do not override vanilla fallen-empire ship creation effects here.
# The old version redefined create_fallen_empire_large_ship / create_fallen_empire_small_ship,
# which affected every fallen empire that used vanilla starting navy generation.

awaken_fallen_machine_empire = {
	every_country = {
		limit = {
			is_country_type = fallen_empire
			is_machine_empire = yes
			not = { has_country_flag = fallen_empire_machine_LgateGrander }
		}

		country_event = { id = fallen_machine_empire.1 days = 30 random = 30 }
	}
}

# Scope: planet
# Root: the fallen empire country in fallen_empire_init initializers.
# Creates a GateGrander/Nanite-only starting fleet without touching vanilla FE navy effects.
create_FEI_lgategrander_starting_navy = {
	create_fleet = {
		name = nanite_swarmer
		effect = {
			set_owner = root
			set_location = prev
			while = {
				count = 4
				create_ship = {
					name = NAME_Nanite_Mothership
					design = NAME_Nanite_Mothership
				}
				while = {
					count = 8
					create_ship = {
						name = NAME_Nanite_Interdictor
						design = NAME_Nanite_Interdictor
					}
				}
			}
			add_modifier = { modifier = gray_warship }
		}
	}
}
