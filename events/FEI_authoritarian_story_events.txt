# Hidden setup: create or extend the vanilla Shroudwalker enclave in this system.
country_event = {
	id = fei_authoritarian_story.19
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = FEI_authoritarian_veil_machine_excavator
		exists = event_target:FEI_authoritarian_veil_system
		exists = event_target:FEI_authoritarian_veil_machine_planet
		NOT = { has_global_flag = FEI_authoritarian_shroudwalker_enclave_arrived }
	}

	immediate = {
		set_global_flag = FEI_authoritarian_shroudwalker_enclave_arrived
		event_target:FEI_authoritarian_veil_system = {
			random_system_planet = {
				limit = {
					is_star = no
					NOT = { is_planet_class = pc_neutron_star }
				}
				set_planet_flag = shroudwalker_enclave_planet
				save_global_event_target_as = shroudwalker_enclave_planet
				if = {
					limit = { NOT = { exists = event_target:shroudwalker_enclave_country } }
					create_species = {
						homeworld = this
						name = random
						class = SHROUDWALKER
						namelist = NECROID1
						portrait = random
						traits = {
							ideal_planet_class = pc_habitat
							trait = trait_psionic
							trait = random_traits
						}
					}
					last_created_species = {
						save_event_target_as = shroudwalker_enclave_species
					}
					create_shroudwalker_enclave_country = yes
					solar_system = {
						save_global_event_target_as = shroudwalker_enclave_system
					}
				}
				else = {
					event_target:shroudwalker_enclave_country = {
						create_shroudwalker_enclave_starbase = yes
					}
				}
				add_deposit = d_zro_deposit_5
			}
		}
		event_target:shroudwalker_enclave_country = {
			establish_communications_no_message = root
		}
		country_event = { id = fei_authoritarian_story.20 days = 30 }
	}
}

# Hidden setup: create or refresh the friendly neutral guard country and anchor the deep-space citadel.
country_event = {
	id = fei_authoritarian_story.29
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = FEI_authoritarian_veil_machine_excavator
		exists = event_target:FEI_authoritarian_veil_machine_planet
	}

	immediate = {
		if = {
			limit = { NOT = { exists = event_target:FEI_authoritarian_neutral_guard_country } }
			create_country = {
				name = NAME_FEI_authoritarian_neutral_guard_country
				type = faction
				auto_delete = no
				flag = random
				effect = {
					set_country_flag = FEI_authoritarian_neutral_guard_country
					save_global_event_target_as = FEI_authoritarian_neutral_guard_country
					set_graphical_culture = fallen_empire_03
					add_global_ship_design = "NAME_Alpha"
					add_global_ship_design = "NAME_Beta"
					add_global_ship_design = "NAME_Gamma"
					add_global_ship_design = "NAME_FEI_authoritarian_flagship"
					add_global_ship_design = "NAME_FEI_authoritarian_deep_space_citadel"
				}
			}
		}

		event_target:FEI_authoritarian_neutral_guard_country = {
			establish_communications_no_message = root
			add_modifier = { modifier = fei_authoritarian_neutral_guard_combat_boost days = -1 }
			add_global_ship_design = "NAME_Alpha"
			add_global_ship_design = "NAME_Beta"
			add_global_ship_design = "NAME_Gamma"
			add_global_ship_design = "NAME_FEI_authoritarian_flagship"
			add_global_ship_design = "NAME_FEI_authoritarian_deep_space_citadel"
			set_faction_hostility = { target = root set_friendly = yes }
			if = {
				limit = { exists = event_target:FEI_authoritarian_imperium }
				set_faction_hostility = { target = event_target:FEI_authoritarian_imperium set_friendly = yes }
			}
			if = {
				limit = { exists = event_target:curator_enclave_country }
				set_faction_hostility = { target = event_target:curator_enclave_country set_friendly = yes }
			}
			if = {
				limit = { exists = event_target:shroudwalker_enclave_country }
				set_faction_hostility = { target = event_target:shroudwalker_enclave_country set_friendly = yes }
			}
		}
		if = {
			limit = { exists = event_target:curator_enclave_country }
			event_target:curator_enclave_country = {
				establish_communications_no_message = event_target:FEI_authoritarian_neutral_guard_country
				set_faction_hostility = { target = event_target:FEI_authoritarian_neutral_guard_country set_friendly = yes }
			}
		}
		if = {
			limit = { exists = event_target:shroudwalker_enclave_country }
			event_target:shroudwalker_enclave_country = {
				establish_communications_no_message = event_target:FEI_authoritarian_neutral_guard_country
				set_faction_hostility = { target = event_target:FEI_authoritarian_neutral_guard_country set_friendly = yes }
			}
		}
		if = {
			limit = { exists = event_target:FEI_authoritarian_imperium }
			event_target:FEI_authoritarian_imperium = {
				establish_communications_no_message = event_target:FEI_authoritarian_neutral_guard_country
				set_faction_hostility = { target = event_target:FEI_authoritarian_neutral_guard_country set_friendly = yes }
			}
		}

		if = {
			limit = {
				event_target:FEI_authoritarian_neutral_guard_country = {
					NOT = { any_owned_fleet = { has_fleet_flag = FEI_authoritarian_deep_space_citadel_fleet } }
				}
			}
			event_target:FEI_authoritarian_neutral_guard_country = {
				create_fleet = {
					name = NAME_FEI_authoritarian_deep_space_citadel_fleet
					settings = { spawn_debris = no }
					effect = {
						set_owner = event_target:FEI_authoritarian_neutral_guard_country
						create_ship = {
							name = NAME_FEI_authoritarian_deep_space_citadel
							design = "NAME_FEI_authoritarian_deep_space_citadel"
							graphical_culture = fallen_empire_03
							upgradable = no
						}
						set_location = {
							target = event_target:FEI_authoritarian_veil_machine_planet
							distance = 10
							angle = 0
						}
						set_fleet_flag = FEI_authoritarian_deep_space_citadel_fleet
						save_global_event_target_as = FEI_authoritarian_deep_space_citadel_fleet_target
						set_fleet_stance = passive
						set_aggro_range_measure_from = self
						set_aggro_range = 0
					}
				}
			}
		}
		set_global_flag = FEI_authoritarian_deep_space_citadel_spawned
	}
}

# Hidden dispatcher: pick one surviving non-militarist, non-authoritarian fallen empire at a time.

	id = fei_authoritarian_story.30
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = FEI_authoritarian_veil_machine_excavator
		has_global_flag = FEI_authoritarian_veil_city_state_spawned
	}

	immediate = {
		if = {
			limit = {
				any_country = {
					OR = {
						is_country_type = fallen_empire
						is_country_type = awakened_fallen_empire
					}
					NOT = { has_country_flag = FEI_militarist_forerunner_empire }
					NOT = { has_country_flag = FEI_authoritarian_fallen_empire }
					NOT = { has_country_flag = FEI_authoritarian_legacy_aid_contacted }
				}
			}
			random_country = {
				limit = {
					OR = {
						is_country_type = fallen_empire
						is_country_type = awakened_fallen_empire
					}
					NOT = { has_country_flag = FEI_militarist_forerunner_empire }
					NOT = { has_country_flag = FEI_authoritarian_fallen_empire }
					NOT = { has_country_flag = FEI_authoritarian_legacy_aid_contacted }
				}
				set_country_flag = FEI_authoritarian_legacy_aid_contacted
				save_global_event_target_as = FEI_authoritarian_current_legacy_helper
			}
			country_event = { id = fei_authoritarian_story.31 days = 30 }
		}
		else = {
			country_event = { id = fei_authoritarian_story.40 days = 540 random = 180 }
		}
	}
