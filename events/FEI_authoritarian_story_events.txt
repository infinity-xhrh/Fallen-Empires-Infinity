# Hidden setup: create or extend the vanilla Shroudwalker enclave in this system.
country_event = {
	id = fei_authoritarian_story.19
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = FEI_authoritarian_veil_machine_excavator
		exists = event_target:FEI_authoritarian_veil_system
		exists = event_target:FEI_authoritarian_veil_machine_planet
		NOT = { has_global_flag = FEI_authoritarian_shroudwalker_enclave_arrived }
	}

	immediate = {
		set_global_flag = FEI_authoritarian_shroudwalker_enclave_arrived
		event_target:FEI_authoritarian_veil_system = {
			random_system_planet = {
				limit = {
					is_star = no
					NOT = { is_planet_class = pc_neutron_star }
				}
				set_planet_flag = shroudwalker_enclave_planet
				save_global_event_target_as = shroudwalker_enclave_planet
				if = {
					limit = { NOT = { exists = event_target:shroudwalker_enclave_country } }
					create_species = {
						homeworld = this
						name = random
						class = SHROUDWALKER
						namelist = NECROID1
						portrait = random
						traits = {
							ideal_planet_class = pc_habitat
							trait = trait_psionic
							trait = random_traits
						}
					}
					last_created_species = {
						save_event_target_as = shroudwalker_enclave_species
					}
					create_shroudwalker_enclave_country = yes
					solar_system = {
						save_global_event_target_as = shroudwalker_enclave_system
					}
				}
				else = {
					event_target:shroudwalker_enclave_country = {
						create_shroudwalker_enclave_starbase = yes
					}
				}
				add_deposit = d_zro_deposit_5
			}
		}
		event_target:shroudwalker_enclave_country = {
			establish_communications_no_message = root
		}
		country_event = { id = fei_authoritarian_story.20 days = 30 }
	}
}

# Hidden dispatcher: pick one surviving non-militarist, non-authoritarian fallen empire at a time.
country_event = {
	id = fei_authoritarian_story.30
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = FEI_authoritarian_veil_machine_excavator
		has_global_flag = FEI_authoritarian_veil_city_state_spawned
	}

	immediate = {
		if = {
			limit = {
				any_country = {
					OR = {
						is_country_type = fallen_empire
						is_country_type = awakened_fallen_empire
					}
					NOT = { has_country_flag = FEI_militarist_forerunner_empire }
					NOT = { has_country_flag = FEI_authoritarian_fallen_empire }
					NOT = { has_country_flag = FEI_authoritarian_legacy_aid_contacted }
				}
			}
			random_country = {
				limit = {
					OR = {
						is_country_type = fallen_empire
						is_country_type = awakened_fallen_empire
					}
					NOT = { has_country_flag = FEI_militarist_forerunner_empire }
					NOT = { has_country_flag = FEI_authoritarian_fallen_empire }
					NOT = { has_country_flag = FEI_authoritarian_legacy_aid_contacted }
				}
				set_country_flag = FEI_authoritarian_legacy_aid_contacted
				save_global_event_target_as = FEI_authoritarian_current_legacy_helper
			}
			country_event = { id = fei_authoritarian_story.31 days = 30 }
		}
		else = {
			country_event = { id = fei_authoritarian_story.40 days = 540 random = 180 }
		}
	}
}
