country_event = {
	id = FEI_settings.1
	title = FEI_settings.1.name
	desc = FEI_settings.1.desc
	is_triggered_only = yes
	picture = GFX_evt_fallen_empire
	event_picture_background = GFX_evt_fallen_empire
	picture_event_data = {
		portrait = root.ruler
		room = root
		planet_background = root.capital_scope
	}

	trigger = {
		is_ai = no
	}

	immediate = {
		set_country_flag = FEI_settings_panel_seen
		if = {
			limit = {
				NOR = {
					has_global_flag = FEI_settings_difficulty_reduced
					has_global_flag = FEI_settings_difficulty_standard
					has_global_flag = FEI_settings_difficulty_hard
					has_global_flag = FEI_settings_difficulty_elite
					has_global_flag = FEI_settings_difficulty_nightmare
				}
			}
			set_global_flag = FEI_settings_difficulty_standard
		}
		if = {
			limit = {
				NOR = {
					has_global_flag = FEI_settings_beauty_mode_enabled
					has_global_flag = FEI_settings_beauty_mode_disabled
				}
			}
			set_global_flag = FEI_settings_beauty_mode_disabled
		}
	}

	option = { name = FEI_settings.1.update country_event = { id = FEI_settings.2 } }
	option = { name = FEI_settings.1.difficulty country_event = { id = FEI_settings.3 } }
	option = { name = FEI_settings.1.beauty country_event = { id = FEI_settings.4 } }
	option = { name = FEI_settings.1.intro country_event = { id = FEI_settings.5 } }
	option = { name = FEI_settings.1.tutorial country_event = { id = FEI_settings.6 } }
	option = { name = FEI_settings.close }
}

country_event = {
	id = FEI_settings.20
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			has_country_flag = fallen_empire_machine_LgateGrander
			has_country_flag = FEI_militarist_forerunner_empire
			has_country_flag = FEI_authoritarian_fallen_empire
			AND = {
				exists = event_target:FEU_country
				is_same_value = event_target:FEU_country
			}
		}
	}

	immediate = {
		remove_modifier = fei_settings_difficulty_reduced
		remove_modifier = fei_settings_difficulty_hard
		remove_modifier = fei_settings_difficulty_elite
		remove_modifier = fei_settings_difficulty_nightmare
		if = { limit = { has_global_flag = FEI_settings_difficulty_reduced } add_modifier = { modifier = fei_settings_difficulty_reduced days = -1 } }
		if = { limit = { has_global_flag = FEI_settings_difficulty_hard } add_modifier = { modifier = fei_settings_difficulty_hard days = -1 } }
		if = { limit = { has_global_flag = FEI_settings_difficulty_elite } add_modifier = { modifier = fei_settings_difficulty_elite days = -1 } }
		if = { limit = { has_global_flag = FEI_settings_difficulty_nightmare } add_modifier = { modifier = fei_settings_difficulty_nightmare days = -1 } }
	}
}
