namespace = fei_militarist_story

# Global target recovery.
# In earlier builds the militarist forerunner was only saved as a local event target in the initializer.
# This hidden monthly event repairs that state by locating the country flag.
country_event = {
	id = fei_militarist_story.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { exists = event_target:FEI_militarist_forerunner }
		any_country = {
			has_country_flag = FEI_militarist_forerunner_empire
		}
	}

	immediate = {
		random_country = {
			limit = {
				has_country_flag = FEI_militarist_forerunner_empire
			}
			save_global_event_target_as = FEI_militarist_forerunner
		}
	}
}


# Monthly scanner: if any normal country has established contact with the militarist fallen empire,
# open the delayed wormhole early if needed, then start the active contact chain once.
country_event = {
	id = fei_militarist_story.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = default
		NOT = { has_country_flag = FEI_militarist_forerunner_empire }
		NOT = { has_global_flag = fei_militarist_forerunner_returned }
		NOT = { has_global_flag = fei_militarist_active_contact_started }
		exists = event_target:FEI_militarist_forerunner
		NOT = { is_same_value = event_target:FEI_militarist_forerunner }
		has_established_contact = event_target:FEI_militarist_forerunner
	}

	immediate = {
		set_global_flag = fei_militarist_active_contact_started
		save_global_event_target_as = fei_militarist_contacted_country
		if = {
			limit = {
				NOT = { has_global_flag = FEI_militarist_wormhole_opened }
				exists = event_target:FEI_militarist_forerunner
			}
			event_target:FEI_militarist_forerunner = {
				country_event = { id = fei_initialize.7 }
			}
		}
		country_event = { id = fei_militarist_story.10 days = 5 }
	}
}

					establish_communications_no_message = root
					create_leader = {
						class = commander

						skip_background_generation = yes
						immortal = yes
						event_leader = yes

						skill = 10

							1 = subclass_commander_admiral
							2 = leader_trait_aggressive_2
							3 = leader_trait_cautious_2

					}
				}
			}
		}
		country_event = { id = fei_militarist_story.71 days = 5 }
	}

	option = { name = OK }

# Elu's destiny trait: intense battle teaching counter.
# This = owner of fleet 1 (winner)
# From = owner of fleet 2 (loser)
# FromFrom = fleet 1
# FromFromFrom = fleet 2
country_event = {
	id = fei_militarist_story.900
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		fromfrom = {
			exists = leader
			leader = {
				has_trait = leader_trait_fei_elu_destiny
				NOT = { has_leader_flag = fei_elu_master_gunner_teaching_completed }
			}
		}
		fromfromfrom = {
			fleet_power >= 10000
		}
		any_owned_leader = {
			leader_class = commander
			has_trait = leader_trait_master_gunner
			NOT = { has_trait = leader_trait_fei_elu_master_gunner }
		}
	}

	immediate = {
		fromfrom = {
			leader = {
				if = {
					limit = { NOT = { is_variable_set = fei_elu_intense_battle_count } }
					set_variable = { which = fei_elu_intense_battle_count value = 0 }
				}
				change_variable = { which = fei_elu_intense_battle_count value = 1 }
				if = {
					limit = {
						check_variable = { which = fei_elu_intense_battle_count value >= 5 }
						owner = {
							any_owned_leader = {
								leader_class = commander
								has_trait = leader_trait_master_gunner
								NOT = { has_trait = leader_trait_fei_elu_master_gunner }
							}
						}
					}
					set_leader_flag = fei_elu_master_gunner_teaching_completed
					owner = { country_event = { id = fei_militarist_story.901 } }
				}
			}
		}
	}
}
