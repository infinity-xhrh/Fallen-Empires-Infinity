        }
    }
    fire_only_once = yes
    immediate = {
        random_list = {
            2 = {}
            1 = {
                no_scope = {
                    spawn_system = {
                        initializer = FEU_capital_system
                        min_distance >= 600
                        max_distance <= 650
                        min_orientation_angle = 180
                        max_orientation_angle = 90
                        hyperlane = no
                    }
                    random_system = {
                        limit = {
                            any_neighbor_system = {
                                not = { exists = space_owner }
                            }
                        }
                        spawn_system = {
                            initializer = FEU_outer_system
                            min_jumps = 1
                            max_jumps = 1
                            hyperlane = yes
                        }
                        last_created_system = {
                            system_event = { id = FEU_main_event.3 days = 1 }
                            spawn_natural_wormhole = {
                                bypass_type = wormhole
                                random_pos = no
                                orbit_angle = 360
                            }
                            link_wormholes = event_target:FEU_capital_system
                        }
                    }
                }
            }
        }
    }
}

#老登阻止天堂之战
country_event = {
    id = FEU_main_event.8
    is_triggered_only = yes
    trigger = {
        is_fallen_empire_utpiod = yes
    }
    hide_window = yes
    immediate = {
        country_event = { id = FEU_main_event.8 years = 1 }
        if = {
            limit = {
                has_global_flag = war_in_heaven_ongoing
            }
            country_event = { id = FEU_main_event.100 years = 1 }
            set_country_type = awakened_fallen_empire
            change_government = {
                civics = {
                    civic = civic_revanchist_fervor
                    civic = civic_ancient_caches_of_technology
                }
            }
            if = {
                limit = { exists = event_target:FirstSleeper }
                declare_war = {
                    attacker_war_goal = wg_end_threat_crisis
                    target = event_target:FirstSleeper
                }
            }
            if = {
                limit = { exists = event_target:SecondSleeper }
                declare_war = {
                    attacker_war_goal = wg_end_threat_crisis
                    target = event_target:SecondSleeper
                }
            }
            while = {
                count = 8
                random_owned_planet = { create_fe_utpoid_fleet = yes }
            }
            every_playable_country = {
                country_event = { id = FEU_main_event.9 }
            }
            capital_scope.solar_system = {
                spawn_megastructure = {
                    owner = event_target:FEU_country
                    type = mega_shipyard_3
                }
            }
        }
    }
}

#给觉醒老登刷船  年检
country_event = {
    id = FEU_main_event.100
    is_triggered_only = yes
    hide_window = yes
    trigger = {
        is_fallen_empire_utpiod_awaked = yes
    }
    immediate = {
        country_event = { id = FEU_main_event.100 months = 6 }
        export_trigger_value_to_variable = {
            trigger = count_controlled_fleet
            variable = FEU_fleet_counts
        }
        if = {
            limit = {
                check_variable = {
                    which = FEU_fleet_counts value <= 3
                }
            }
            random_owned_planet = { create_fe_utpoid_fleet = yes }
            random_owned_planet = { create_fe_utpoid_fleet = yes }
        }
        if = {
            limit = {
                is_at_war = yes
            }
            while = {
                count = 4
                random_owned_planet = { create_fe_utpoid_fleet = yes }
            }
        }
    }
}
#觉醒老登给你送船
country_event = {
    id = FEU_main_event.101
    title = FEU_main_event.4.title
    diplomatic = yes
    is_triggered_only = yes
    picture_event_data = {
        planet_background = root.capital_scope
        room = root
        portrait = event_target:FEU_country
    }
    trigger = {
        OR = {
            AND = {
                exists = event_target:FirstSleeper
                is_at_war_with = event_target:FirstSleeper
            }
            AND = {
                exists = event_target:SecondSleeper
                is_at_war_with = event_target:SecondSleeper
            }
        }
    }
    desc = FEU_main_event.101.desc
    option = {
        name = OK
        capital_scope = {
            create_fleet = {
                name = NAME_utpoid_fleet
                effect = {
                    set_owner = root
                    set_location = prev
                    create_ship = {
                        name = random
                        design = NAME_utpoid_keeper
                    }
                    while = {
                        count = 12
                        create_ship = {
                            name = random
                            design = NAME_utpoid_protecter
                        }
                    }
                    while = {
                        count = 30
                        create_ship = {
                            name = random
                            design = NAME_utpoid_attacker
                        }
                    }
                }
            }
        }
    }
    immediate = {
        country_event = { id = FEU_main_event.101 months = 8 random = 2 }
    }
}
