namespace = fei_initialize

#军国老登的星系
event = {
	id = fei_initialize.1
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_spawn_system_batch = begin
		random_system = {
			limit = {
				is_rim_system = no
				not = { has_star_flag = crisis_spawn_exclude }
			}
			spawn_system = {
				min_distance >= 15
				max_distance <= 25
				initializer = FEI_initializer_hepingshichuang
				hyperlane = yes
			}
		}
		random_rim_system = {
			spawn_system = {
				min_distance >= 50
				max_distance <= 100
				direction = rimwards
				initializer = FEI_initializer_yanwuchang
				hyperlane = no
			}
		}
		set_spawn_system_batch = end
		if = {
			limit = { exists = event_target:FEI_militarist_forerunner }
			event_target:FEI_militarist_forerunner = {
				country_event = { id = fei_initialize.7 days = 18000 }
			}
		}
	}
}

# 军国堕落特色陆军地面战逻辑
# 进入地面战后启动两个星球作用域循环：
# 1. 超时空军团每 10 天按前线数量抹杀进攻方部队。
# 2. 心灵专家与增殖机器人每 10 天生成防守增援。
planet_event = {
    id = fei_initialize.2
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        has_ground_combat = yes
        exists = owner
        OR = {
            any_ground_combat_defender = { army_type = FEI_chrono_army }
            any_ground_combat_defender = { army_type = FEI_psi_army }
            any_ground_combat_defender = { army_type = FEI_duplicant_army }
        }
    }

    immediate = {
        if = {
            limit = {
                NOT = { has_planet_flag = FEI_chrono_army_triggered }
                any_ground_combat_defender = { army_type = FEI_chrono_army }
            }
            planet_event = { id = fei_initialize.4 days = 10 }
            set_timed_planet_flag = {
                flag = FEI_chrono_army_triggered
                days = 10
            }
        }
        if = {
            limit = {
                NOT = { has_planet_flag = FEI_special_army_reinforce_triggered }
                OR = {
                    any_ground_combat_defender = { army_type = FEI_psi_army }
                    any_ground_combat_defender = { army_type = FEI_duplicant_army }
                }
            }
            planet_event = { id = fei_initialize.6 days = 10 }
            set_timed_planet_flag = {
                flag = FEI_special_army_reinforce_triggered
                days = 10
            }
        }
    }
}

# 超时空军团：每 10 天按前线超时空军团数量抹杀进攻方陆军。
planet_event = {
    id = fei_initialize.4
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        has_ground_combat = yes
        any_ground_combat_defender = {
            army_type = FEI_chrono_army
        }
    }

    immediate = {
        planet_event = { id = fei_initialize.4 days = 10 }
        set_timed_planet_flag = {
            flag = FEI_chrono_army_triggered
            days = 10
        }
        export_trigger_value_to_variable = {
            trigger = count_ground_combat_defender
            parameters = {
                limit = {
                    is_in_frontline = yes
                    army_type = FEI_chrono_army
                }
            }
            variable = FEI_chrono_army_counter
        }
        while = {
            count = FEI_chrono_army_counter
            if = {
                limit = {
                    any_ground_combat_attacker = { always = yes }
                }
                random_ground_combat_attacker = {
                    remove_army = yes
                }
            }
        }
    }
}

# 心灵专家 / 增殖机器人：改为星球作用域循环，避免 on_army_killed_in_combat 的跨作用域不稳定。
# 每 10 天在战斗仍持续时，为仍在前线的特色部队生成一支防守增援。
planet_event = {
    id = fei_initialize.6
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        has_ground_combat = yes
        exists = owner
        OR = {
            any_ground_combat_defender = {
                is_in_frontline = yes
                army_type = FEI_psi_army
            }
            any_ground_combat_defender = {
                is_in_frontline = yes
                army_type = FEI_duplicant_army
            }
        }
    }

    immediate = {
        planet_event = { id = fei_initialize.6 days = 10 }
        set_timed_planet_flag = {
            flag = FEI_special_army_reinforce_triggered
            days = 10
        }
        if = {
            limit = {
                any_ground_combat_defender = {
                    is_in_frontline = yes
                    army_type = FEI_psi_army
                }
            }
            create_army = {
                owner = owner
                type = FEI_psi_army
                species = owner_main_species
            }
        }
        if = {
            limit = {
                any_ground_combat_defender = {
                    is_in_frontline = yes
                    army_type = FEI_duplicant_army
                }
            }
            create_army = {
                owner = owner
                type = FEI_duplicant_army
            }
        }
    }
}

# 创建后 50 年开启演武场与和平视窗之间的虫洞；若玩家提前建立通讯，则由接触扫描事件提前触发。
# ROOT is the militarist fallen empire country.
country_event = {
	id = fei_initialize.7
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		has_country_flag = FEI_militarist_forerunner_empire
		NOT = { has_global_flag = FEI_militarist_wormhole_opened }
		exists = event_target:FEI_GXXLZ_system_pulsar
		exists = event_target:FEI_GXXLZ_home_system
	}

	immediate = {
		set_global_flag = FEI_militarist_wormhole_opened
		event_target:FEI_GXXLZ_system_pulsar = {
			spawn_natural_wormhole = {
				bypass_type = wormhole
				random_pos = yes
				orbit_angle = 360
			}
		}
		event_target:FEI_GXXLZ_home_system = {
			spawn_natural_wormhole = {
				bypass_type = wormhole
				random_pos = yes
				orbit_angle = 360
			}
			link_wormholes = event_target:FEI_GXXLZ_system_pulsar
		}
	}
}
